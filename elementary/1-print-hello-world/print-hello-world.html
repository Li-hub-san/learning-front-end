Greet the World!
// 1. Write a program that prints ‘Hello World’ to the screen.

function printGreeting() {
  updatePrintElText("Hello", "World!");
}

function updatePrintElText(greeting, who) {
  const printEl = getElement("output");
  printEl.innerHTML =
          `<p><span class="greeting">${greeting} </span>
           <span class="who">${who}</span></p>`;
}
/*targets*/
span.greeting {
  animation-name: greet, pop;
  animation-delay: 0s, .4s;
  animation-fill-mode: forwards;
  animation-duration: 200ms, 1.5s;
  animation-iteration-count: 1;
  position: relative;
}

span.who {
  font-size: initial;
  animation-name: world, toggleOpacity;
  animation-fill-mode: forwards;
  animation-duration: 200ms, 1.9s;
  animation-iteration-count: 1;
  position: relative;
}

/*animations*/
